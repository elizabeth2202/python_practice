%matplotlib inline
import numpy as np
import matplotlib.pyplot as plt

def f(x):
    result = np.sin(x / 5.0) * np.exp(x / 10.0) + 5 * np.exp(-x / 2.0)
    return result

x = np.linspace(1, 15, 50) #  50   1  15 
y = f(x)
plt.plot(x, y)
plt.ylabel('f(x)')
plt.xlabel('x')
plt.show()
